%% 1.3 Defining commands used in document

% Header
\newcommand{\name}[1]{%
  \centerline{\Huge{#1}}
}

% Contact Line
\newcommand\contact[5]{%
    \centerline{%
        \makebox[0pt][c]{%
            #1 {\large\textperiodcentered} #2 {\large\textperiodcentered} #3
            {\large\textperiodcentered} #4 {\large\textperiodcentered} #5%
        }%
    }%
}

% Section
\newcommand{\datedsection}[2]{%
  \section[#1]{#1 \hfill #2}%
}

% Subsection
\newcommand{\NameDate}[2]{%
  \subsection[#1]{#1 \hfill #2}%
}
 
% Job Title 
\newcommand{\Jobtitle}[2]{%
    \textit{#1} \hfill \textit{#2} \\[-6.25mm]%
}

% Job Title if you don't use bullets for that entry
\newcommand{\JobtitleNoBullets}[2]{%
    \textit{#1} \hfill \textit{#2} \\[-2.5mm]%
}

% Bullet(-design)
\newcommand{\myitem}{\item[\tiny$\blacksquare$]}

\setlist{leftmargin=0.5cm}

% Bullets
\newcommand{\onebullets}[1]{%
    \begin{itemize}
        \setlength\itemsep{-0.35em}
        \myitem #1
    \end{itemize}%
}

\newcommand{\twobullets}[2]{%
    \begin{itemize}
        \setlength\itemsep{-0.35em}
        \myitem #1
        \myitem #2
    \end{itemize}%
}

\newcommand{\threebullets}[3]{%
    \begin{itemize}
        \setlength\itemsep{-0.35em}
        \myitem #1
        \myitem #2
        \myitem #3
    \end{itemize}%
}



%% Syntax follows the one above.  Commands are used when document begins.
